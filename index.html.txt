function adicionarCarrinho(){

let peso=document.getElementById("peso")

let preco=peso.value

let pesoTxt=peso.options[peso.selectedIndex].text

let sabor=document.getElementById("sabor")?.value||""

let casca=document.getElementById("casca")?.value||""

let emb=document.getElementById("emb").value

let qtd=document.getElementById("qtd").value

let item={tipoAtual,sabor,casca,pesoTxt,preco,qtd,emb}

carrinho.push(item)

renderCarrinho()

}

function renderCarrinho(){

let html=""

let total=0

carrinho.forEach((item,i)=>{

let sub=item.preco*item.qtd

total+=sub

html+=`

<div>

${item.qtd}x ${item.tipoAtual} ${item.pesoTxt}<br>

${item.sabor? "Sabor:"+item.sabor+"<br>":""}

${item.casca? "Casca:"+item.casca+"<br>":""}

Embalagem:${item.emb}<br>

R$${sub}

<button onclick="remover(${i})">Remover</button>

</div>

<hr>

`

})

document.getElementById("listaCarrinho").innerHTML=html

atualizarTotal()

}

function atualizarTotal(){

let total=0

carrinho.forEach(i=>{

total+=i.preco*i.qtd

})

let entrega=document.getElementById("entrega").value

total+=Number(entrega)

document.getElementById("total").innerText="Total: R$"+total

}

function remover(i){

carrinho.splice(i,1)

renderCarrinho()

}
